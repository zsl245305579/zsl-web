
1.最大线程数，打开的太低，需要增加线程数
vi /etc/security/limits.d/20-nproc.conf
vi /etc/security/limits.d/90-nproc.conf
* soft nproc 2048


[root@elk ~]# ulimit -SHn 65536
注：
	-S 设置软件资源限制
	-H 设置硬件资源限制
	-n 设置内核可以同时可以打开文件描述符

#在文件最后添加
[root@elk ~]# vi /etc/security/limits.conf
* soft nofile 65536
* hard nofile 131072
* soft nproc 2048
* hard nproc 4096

分为3中类型type（有 soft，hard 和 -）
		soft是指当前系统生效的设置值
		hard 系统设置的最大值
-同时设置了soft和hard的值
nofile - 打开文件的最大数目
noproc - 进程的最大数目
soft<=hard soft的限制不能比hard限制高


2.打开虚拟内存的个数太少需要增加

#在文件末尾添加
[root@elk ~]# vi /etc/sysctl.conf
vm.max_map_count=655360

[root@elk ~]#sysctl -p
注：
vm.max_map_count文件允许max_map_count限制虚拟内存的数量

input {
  redis {
	host => "192.168.1.63"
	port => "6379"
	data_type => "list"
	key => "nginx_access"
	type => "redis-input"
  }
}
filter {
	if[type] =~ "nginx_access" {
		grok {
            		patterns_dir => ["/data/application/logstash-5.2.0/vendor/bundle/jruby/1.9/gems/logstash-patterns-core-4.0.2/patterns/"]
            		match => {
                		"message" => "%{NGINXACCESS}"
            		}
        	}
    		date {
        		match => ["timestamp" , "dd/MMM/YYYY:HH:mm:ss Z" ]
    		}
	}
}
output {
     elasticsearch {
      hosts => "192.168.1.63:9200"
      index => "nginx-access-%{+yyyy.MM.dd}"
      }
}
